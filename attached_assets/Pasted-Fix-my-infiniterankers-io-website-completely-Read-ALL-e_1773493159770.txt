export function getSEOForRoute(url: string): SEOMeta | null {
  const path = url.split("?")[0].split("#")[0];
  if (STATIC_PAGES[path]) return STATIC_PAGES[path];
  const slug = path.replace(/^\//, "").replace(/\/$/, "");
  if (STATIC_PAGES["/" + slug]) return STATIC_PAGES["/" + slug];
  if (STATIC_PAGES["/" + slug + "/"]) return STATIC_PAGES["/" + slug + "/"];
  const blogMatch = path.match(/^\/blog\/(.+)$/);
  if (blogMatch && BLOG_POSTS[blogMatch[1]]) return BLOG_POSTS[blogMatch[1]];
  return {
    title: "Infinite Rankers | AI Revenue Growth & Digital Marketing Agency USA",
    description: "AI-powered digital marketing agency delivering measurable revenue growth. SEO, Google Ads, AI automation for USA businesses. Google Partner certified.",
    canonical: `${BASE}${path}`
  };
}

function escapeHtml(s: string): string {
  return s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
}
function escapeAttr(s: string): string {
  return s.replace(/&/g, "&amp;").replace(/"/g, "&quot;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
}

export function injectSEO(html: string, url: string): string {
  const seo = getSEOForRoute(url);
  if (!seo) return html;
  const headEnd = html.indexOf("</head>");
  if (headEnd === -1) return html;
  const tags: string[] = [];
  let result = html;
  const titleTag = `<title>${escapeHtml(seo.title)}</title>`;
  if (/<title>[^<]*<\/title>/.test(result)) {
    result = result.replace(/<title>[^<]*<\/title>/, titleTag);
  } else {
    tags.push(titleTag);
  }
  if (!/<meta\s+name="description"/.test(result)) tags.push(`<meta name="description" content="${escapeAttr(seo.description)}" />`);
  if (!/<link\s+rel="canonical"/.test(result)) tags.push(`<link rel="canonical" href="${escapeAttr(seo.canonical)}" />`);
  tags.push(`<meta property="og:title" content="${escapeAttr(seo.title)}" />`);
  tags.push(`<meta property="og:description" content="${escapeAttr(seo.description)}" />`);
  tags.push(`<meta property="og:url" content="${escapeAttr(seo.canonical)}" />`);
  tags.push(`<meta property="og:type" content="website" />`);
  tags.push(`<meta property="og:site_name" content="Infinite Rankers" />`);
  tags.push(`<meta property="og:image" content="https://infiniterankers.io/images/og-image.png" />`);
  tags.push(`<meta name="twitter:card" content="summary_large_image" />`);
  tags.push(`<meta name="twitter:title" content="${escapeAttr(seo.title)}" />`);
  tags.push(`<meta name="twitter:description" content="${escapeAttr(seo.description)}" />`);
  tags.push(`<meta name="robots" content="index, follow" />`);
  tags.push(`<script type="application/ld+json">${getStructuredData(url, seo)}</script>`);
  result = result.replace("</head>", tags.map(t => `  ${t}`).join("\n") + "\n</head>");
  return result;
}

import { type Express } from "express";
import fs from "fs";
import path from "path";
import express from "express";
import { injectSEO } from "./seo";

export function serveStatic(app: Express) {
  const distPath = path.resolve(__dirname, "public");
  if (!fs.existsSync(distPath)) {
    throw new Error(`Could not find build directory: ${distPath}`);
  }
  app.use(express.static(distPath, { index: false }));
  let render: ((url: string) => string) | null = null;
  const ssrPath = path.resolve(__dirname, "ssr", "entry-server.cjs");
  if (fs.existsSync(ssrPath)) {
    try {
      const ssrModule = require(ssrPath);
      render = ssrModule.render;
      console.log("✅ SSR module loaded");
    } catch (e) {
      console.warn("⚠️ SSR failed, using CSR:", e);
    }
  } else {
    console.warn("⚠️ SSR file not found — using CSR fallback");
  }
  const indexPath = path.resolve(distPath, "index.html");
  const template = fs.readFileSync(indexPath, "utf-8");
  app.use("*splat", (req, res) => {
    let html = template;
    if (render) {
      try {
        const appHtml = render(req.originalUrl);
        html = html.replace("<!--ssr-outlet-->", appHtml);
      } catch (e) {
        console.error("SSR render error:", e);
        html = html.replace("<!--ssr-outlet-->", "");
      }
    }
    html = injectSEO(html, req.originalUrl);
    res.status(200).set({
      "Content-Type": "text/html",
      "Cache-Control": "no-cache, no-store, must-revalidate",
    }).end(html);
  });
}

import React from "react";
import { renderToString } from "react-dom/server";
import { Router } from "wouter";
import App from "./App";

export function render(url: string): string {
  try {
    const staticLocation = () => [url, () => {}] as [string, (to: string) => void];
    return renderToString(
      <Router hook={staticLocation}>
        <App />
      </Router>
    );
  } catch (e) {
    console.error("SSR render error:", e);
    return "";
  }
}
